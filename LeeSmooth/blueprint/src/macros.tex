% Shared macros for the Lee "Introduction to Smooth Manifolds" (2nd ed., GTM 218) blueprint.
% Kept KaTeX-compatible (no fancy packages) to match the leanblueprint web build.
% Deliberately parallel to LeeRiemannian/blueprint/src/macros.tex so that shared
% notation renders identically across the two Lee projects.

\newcommand{\R}{\mathbb{R}}
\newcommand{\C}{\mathbb{C}}
\newcommand{\Z}{\mathbb{Z}}
\newcommand{\N}{\mathbb{N}}
\newcommand{\Rn}{\R^n}
\newcommand{\Hn}{\mathbb{H}^n}       % upper half-space, the model for manifolds with boundary
\newcommand{\Sph}[1]{\mathbb{S}^{#1}}
\newcommand{\Proj}[1]{\mathbb{RP}^{#1}}

% Smooth structure
\newcommand{\Cinf}{C^\infty}
\newcommand{\Sm}[1]{C^\infty(#1)}    % smooth functions on a manifold
\newcommand{\chart}[2]{(#1, #2)}     % a chart (U, \varphi)
\newcommand{\atlas}{\mathcal{A}}

% Tangent/cotangent spaces and bundles
\newcommand{\T}{T}
\newcommand{\Tp}[1]{T_{#1}}
\newcommand{\Ts}{T^*}
\newcommand{\Tsp}[1]{T^*_{#1}}
\newcommand{\dd}{\mathrm{d}}         % differential / exterior derivative
\newcommand{\dif}[1]{\dd #1}
\newcommand{\pushf}{{}_*}            % pushforward, e.g. F\pushf
\newcommand{\pullb}{{}^*}            % pullback, e.g. F\pullb

% Bundles, sections, fields
\newcommand{\Gam}[1]{\Gamma(#1)}     % smooth sections of a bundle
\newcommand{\XX}[1]{\mathfrak{X}(#1)} % smooth vector fields
\newcommand{\Om}[2]{\Omega^{#1}(#2)} % differential k-forms
\newcommand{\rank}{\operatorname{rank}}

% Lie theory
\newcommand{\Lie}[1]{\operatorname{Lie}(#1)}
\newcommand{\lie}{\mathfrak{g}}
\newcommand{\Ad}{\operatorname{Ad}}
\newcommand{\ad}{\operatorname{ad}}
\newcommand{\Lder}{\mathcal{L}}      % Lie derivative
\newcommand{\GL}[1]{\operatorname{GL}(#1)}
\newcommand{\Exp}{\operatorname{exp}}

% Cohomology
\newcommand{\HdR}[2]{H^{#1}_{\mathrm{dR}}(#2)}
\newcommand{\Ker}{\operatorname{Ker}}
\newcommand{\Img}{\operatorname{Im}}
\newcommand{\supp}{\operatorname{supp}}
\newcommand{\sgn}{\operatorname{sgn}}
\newcommand{\id}{\operatorname{id}}

\newcommand{\inner}[2]{\langle #1, #2 \rangle}
\newcommand{\norm}[1]{\lvert #1 \rvert}

% as a plain no-op placeholder here, matching the LeeRiemannian blueprint.
