\section{Embedded Submanifolds}

\begin{definition}
\label{def:5-28-extra-1}
\lean{IsEmbeddedSubmanifold}
\leanok
Let $M$ be a smooth manifold, possibly with boundary. An \emph{embedded
submanifold} of $M$ is a subset $S\subseteq M$ that is a manifold without
boundary in the subspace topology and has a smooth structure for which the
inclusion $S\hookrightarrow M$ is a smooth embedding. Its codimension is
$\dim M-\dim S$, and $M$ is its ambient manifold. An embedded hypersurface has
codimension $1$. The empty set is an embedded submanifold of every dimension.
\end{definition}

\begin{definition}
\label{def:5-28-extra-2}
\lean{Set}
\leanok
An embedded submanifold $S\subseteq M$ is \emph{properly embedded} if its
inclusion into $M$ is proper.
\end{definition}

\begin{corollary}
\label{cor:5-6}
\lean{IsCompact}
\leanok
\dcref{ch5:5.6}
Every compact embedded submanifold is properly embedded.
\end{corollary}

\begin{proof}
\leanok
For every compact $K\subseteq M$, the Hausdorff property makes $K$ closed.
Thus $K\cap S$ is closed in the compact space $S$, hence compact, so the
inclusion $S\hookrightarrow M$ is proper.
\end{proof}


\section{Slice Charts for Embedded Submanifolds}

\begin{definition}
\label{def:5-29-extra-1}
\lean{Set.SatisfiesLocalSliceCondition}
\leanok
For an open set $U\subseteq\mathbb R^n$ and $0\leq k\leq n$, a
\emph{$k$-slice} of $U$ is a set
\[
 \left\{x\in U:x^{k+1}=c^{k+1},\ldots,x^n=c^n\right\}
\]
for fixed constants $c^{k+1},\ldots,c^n$. Thus an $n$-slice is $U$, and each
$k$-slice is homeomorphic to an open subset of $\mathbb R^k$. Coordinates
other than the last $n-k$ may instead be fixed when specified; translating
coordinates makes all fixed constants zero.

Let $M$ be a smooth $n$-manifold and $(U,\varphi)$ a smooth chart. A subset
$S\subseteq U$ is a $k$-slice of $U$ when $\varphi(S)$ is a $k$-slice of
$\varphi(U)$. A set $S\subseteq M$ satisfies the \emph{local $k$-slice
condition} if every $p\in S$ belongs to a chart domain $U$ for which $S\cap U$
is one $k$-slice. Such a chart and its coordinates are respectively called a
\emph{slice chart} and \emph{slice coordinates} for $S$ in $M$.
\end{definition}



\section{Level Sets}

\begin{definition}
\label{def:5-30-extra-1}
\lean{Set.preimage}
\mathlibok
For a map $\Phi:M\to N$ and $c\in N$, the fiber $\Phi^{-1}(c)$ is the
\emph{level set} of $\Phi$ at $c$. If $N=\mathbb R^k$ and $c=0$, it is the
\emph{zero set} of $\Phi$.
\end{definition}


\section{Immersed Submanifolds}

\begin{definition}
\label{def:5-31-extra-1}
\lean{Manifold.ImmersedSubmanifold}
\leanok
Let $M$ be a smooth manifold, possibly with boundary. An \emph{immersed
submanifold} of $M$ is a subset $S\subseteq M$ equipped with a topology, not
necessarily the subspace topology, that makes it a topological manifold without
boundary, and with a smooth structure for which $S\hookrightarrow M$ is a
smooth immersion. Its codimension is $\dim M-\dim S$. This definition concerns
subsets with injective inclusion and does not include images of noninjective
immersions.

Every embedded submanifold is immersed. Unless qualified otherwise,
\emph{smooth submanifold} means an immersed submanifold, and \emph{smooth
hypersurface} means an immersed codimension-$1$ submanifold. Both embedded and
immersed submanifolds are understood to be smooth.
\end{definition}

\begin{proposition}[Topology of an Immersed Submanifold]
\label{exer:5-20}
\lean{Manifold.ImmersedSubmanifold.subspaceTopology_le_givenTopology_iff_isEmbedding}
\leanok
\dcref{ch5:5.20}
Let $S\subseteq M$ be an immersed submanifold. Every subspace-open subset of
$S$ is open in its given manifold topology, and the converse holds exactly
when $S$ is embedded.
\end{proposition}


\section{Restricting Maps to Submanifolds}

\begin{theorem}[Restricting the Domain of a Smooth Map]
\label{thm:5-27}
\lean{ContMDiff.comp}
\mathlibok
\dcref{ch5:5.27}
Let $M$ and $N$ be smooth manifolds, possibly with boundary. If $F:M\to N$ is
smooth and $S\subseteq M$ is an immersed or embedded submanifold, then
$F|_S:S\to N$ is smooth.
\end{theorem}

\begin{proof}
For the smooth inclusion $\iota:S\hookrightarrow M$, one has
$F|_S=F\circ\iota$, which is smooth.
\end{proof}

\begin{corollary}[Embedded Case]
\label{cor:5-30}
\lean{contMDiff_toSubtype_of_isEmbeddedSubmanifold}
\leanok
\dcref{ch5:5.30}
Let $M$ be a smooth manifold and $S\subseteq M$ an embedded submanifold. Every
smooth map $F:N\to M$ with $F(N)\subseteq S$ is smooth as an $S$-valued map.
\end{corollary}

\begin{proof}
\leanok
\sketch Because $S$ has the subspace topology, the subspace universal property
(Proposition A.17(a)) first makes $F$ continuous as an $S$-valued map.
\end{proof}

\begin{definition}
\label{def:5-32-extra-2}
\lean{contMDiff_codRestrict_opens}
\leanok
Let $M$ be a smooth manifold. An immersed submanifold $S\subseteq M$ is
\emph{weakly embedded} if every smooth map $F:N\to M$ with image in $S$ is
smooth as an $S$-valued map.
\end{definition}

\section{Extending Functions from Submanifolds}

\begin{notation}
\label{not:5-34-extra-1}
\lean{C}
\mathlibok
Let $M$ be a smooth manifold and $S\subseteq M$ a smooth submanifold. The
notation $f\in C^\infty(S)$ means that $f:S\to\mathbb R$ is smooth with
respect to the manifold structure on $S$; it does not assert that $f$ locally
extends to a smooth function on $M$.
\end{notation}

\begin{lemma}[Extension Lemma for Functions on Submanifolds]
\label{lem:5-34}
\lean{properly_embedded_submanifold_exists_global_smooth_extension}
\leanok
\dcref{ch5:5.34}
Let $M$ be a smooth manifold, let $S\subseteq M$ be a smooth submanifold, and
let $f\in C^\infty(S)$.

(a) If $S$ is embedded, there are a neighborhood $U$ of $S$ in $M$ and
$\widetilde f\in C^\infty(U)$ such that $\widetilde f|_S=f$.

(b) If $S$ is properly embedded, one may take $U=M$.
\end{lemma}


\section{The Tangent Space to a Submanifold}

\begin{notation}
\label{not:5-35-extra-1}
\lean{Manifold.submanifoldTangentSpace}
\leanok
Let $M$ be a smooth manifold, possibly with boundary, and let $S\subseteq M$ be
an immersed or embedded submanifold. Let $\iota:S\hookrightarrow M$ be the
inclusion. For $p\in S$,
the injective differential $d\iota_p:T_pS\to T_pM$ sends $v\in T_pS$ to the
derivation $\widetilde v=d\iota_p(v)$ satisfying
\[
\widetilde v f=d\iota_p(v)f=v(f\circ\iota)=v(f|_S).
\]
Identify $T_pS$ with $d\iota_p(T_pS)\subseteq T_pM$.
\end{notation}

\begin{definition}
\label{def:5-35-extra-3}
\lean{BoundaryDefiningFunction}
\leanok
A \emph{boundary defining function} for a smooth manifold with boundary $M$ is
a smooth map $f:M\to[0,\infty)$ such that $f^{-1}(0)=\partial M$ and
$df_p\neq0$ for every $p\in\partial M$. For the closed unit ball
$\overline{\mathbb B}^n$, the function $f(x)=1-|x|^2$ is boundary defining.
\end{definition}


\section{Submanifolds with Boundary}

\begin{definition}
\label{def:5-36-extra-2}
\lean{Set}
\leanok
A \emph{regular domain} in a smooth manifold $M$, possibly with boundary, is a
properly embedded codimension-$0$ submanifold with boundary.
\end{definition}

\begin{definition}
\label{def:5-36-extra-4}
\lean{Set.SatisfiesLocalSliceConditionWithBoundary}
\leanok
Let $M$ be a smooth manifold without boundary and $(U,(x^i))$ a chart. A
\emph{$k$-dimensional half-slice} of $U$ is, for fixed
$c^{k+1},\ldots,c^n$, a set
\[
\left\{x\in U:x^{k+1}=c^{k+1},\ldots,x^n=c^n,\ x^k\geq0\right\}.
\]
A subset $S\subseteq M$ satisfies the \emph{local $k$-slice condition for
submanifolds with boundary} if every point of $S$ lies in a chart domain $U$
such that $S\cap U$ is either a $k$-slice or a $k$-dimensional half-slice. The
corresponding charts are called \emph{interior slice charts} and
\emph{boundary slice charts}, respectively.
\end{definition}


\begin{theorem}[Restricting Maps to Submanifolds with Boundary]
\label{thm:5-53}
\lean{contMDiff_restrict_subtype_of_isEmbeddedSubmanifoldWithBoundary}
\leanok
\dcref{ch5:5.53}
Let $M$ and $N$ be smooth manifolds with boundary, and let $S\subseteq M$ be an
embedded submanifold with boundary.

(a) If $F:M\to N$ is smooth, then $F|_S:S\to N$ is smooth.

(b) If $\partial M=\varnothing$ and $F:N\to M$ is smooth with $F(N)\subseteq
S$, then $F$ is smooth as an $S$-valued map.
\end{theorem}

\begin{remark}
\label{rem:5-36-extra-5}
\lean{Manifold.IsSmoothEmbedding.contMDiff_toSubtype}
\leanok
The hypothesis $\partial M=\varnothing$ in Theorem~\ref{thm:5-53}(b) is
unnecessary.
\end{remark}
