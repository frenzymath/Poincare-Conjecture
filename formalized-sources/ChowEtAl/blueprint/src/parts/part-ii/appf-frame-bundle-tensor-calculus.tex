\chapter{Tensor Calculus on the Frame Bundle}
\label{app:chow-ii-frame-bundle-tensor-calculus}

This appendix records the frame-bundle formalism used for the differential
Harnack calculation.  Let $\mathcal F\mathcal M$ be the general linear frame
bundle of a Riemannian manifold $(\mathcal M^n,g)$, and let
$\mathcal O\mathcal M\subset\mathcal F\mathcal M$ be its orthonormal frame
bundle.  Components of tensors below are functions on the frame bundle.

\section{Tensors and the natural frame}

\begin{definition}[Tensor components on the frame bundle]
\label{def:chow-ii-frame-tensor-components}
\dcref{appf:F.2}
\group{Frame Bundle Calculus}
\source{chow-et-al-ricci-flow-part-ii:page-0439,chow-et-al-ricci-flow-part-ii:page-0440}
For a tensor $U$ of type $(p,q)$ and a frame $Y=\{Y_a\}$, write
$U^{a_1\cdots a_q}_{b_1\cdots b_p}(Y)$ for its components in $Y$.  The
associated vector-valued function is
\[
 U^\dagger:\mathcal F\mathcal M\longrightarrow
 \mathbb R^{n^{p+q}},\qquad
 U^\dagger(Y)=\bigl(U^{a_1\cdots a_q}_{b_1\cdots b_p}(Y)\bigr).
\]
Under $Z_b=G^a{}_bY_a$, $G\in\mathrm{GL}(n,\mathbb R)$, these components obey
\[
 U^{a_1\cdots a_q}_{b_1\cdots b_p}(Z)=
 (G^{-1})^{a_1}{}_{c_1}\cdots(G^{-1})^{a_q}{}_{c_q}
 G^{d_1}{}_{b_1}\cdots G^{d_p}{}_{b_p}
 U^{c_1\cdots c_q}_{d_1\cdots d_p}(Y).
\]
\end{definition}

\begin{definition}[Vertical frame fields]
\label{def:chow-ii-vertical-frame-fields}
\dcref{appf:F.6}
\group{Frame Bundle Calculus}
\source{chow-et-al-ricci-flow-part-ii:page-0443,chow-et-al-ricci-flow-part-ii:page-0444}
For the standard matrix unit $e(a,b)\in\mathfrak{gl}(n,\mathbb R)$, define
\[
 \Lambda^a_b(Y)=(\rho_Y)_*e(a,b).
\]
In local frame coordinates $y_a^i$,
$\Lambda^a_b=y_b^i\,\partial/\partial y_a^i$.
\end{definition}

\begin{lemma}[Vertical action on tensor components]
\label{lem:chow-ii-vertical-tensor-action}
\dcref{appf:F.8}
\group{Frame Bundle Calculus}
\source{chow-et-al-ricci-flow-part-ii:page-0444}
\uses{def:chow-ii-frame-tensor-components,def:chow-ii-vertical-frame-fields}
For a $(p,q)$-tensor,
\[
 \Lambda^c_d U^{a_1\cdots a_q}_{b_1\cdots b_p}
 =\sum_{\ell=1}^p\delta^c_{b_\ell}
 U^{a_1\cdots a_q}_{b_1\cdots b_{\ell-1}d b_{\ell+1}\cdots b_p}
 -\sum_{k=1}^q\delta^{a_k}_d
 U^{a_1\cdots a_{k-1}c a_{k+1}\cdots a_q}_{b_1\cdots b_p}.
\]
\end{lemma}

\begin{definition}[Horizontal frame fields]
\label{def:chow-ii-horizontal-frame-fields}
\dcref{appf:F.9}
\group{Frame Bundle Calculus}
\source{chow-et-al-ricci-flow-part-ii:page-0445,chow-et-al-ricci-flow-part-ii:page-0446}
For $Y\in\mathcal F\mathcal M$, let $\gamma_{Y_a}$ be the parallel transport
of $Y$ along a curve with initial velocity $Y_a$.  Define
$\nabla_a|_Y=(d/dt)|_{0}\gamma_{Y_a}(t)$.  In local coordinates,
\[
 \nabla_a=y_a^j\left(\frac{\partial}{\partial \widetilde x^j}
 -y_b^k\Gamma^i_{kj}\frac{\partial}{\partial y_b^i}\right).
\]
\end{definition}

\begin{lemma}[Horizontal differentiation is covariant differentiation]
\label{lem:chow-ii-horizontal-covariant-differentiation}
\dcref{appf:F.13,appf:F.14}
\group{Frame Bundle Calculus}
\source{chow-et-al-ricci-flow-part-ii:page-0447,chow-et-al-ricci-flow-part-ii:page-0448,chow-et-al-ricci-flow-part-ii:page-0449}
\uses{def:chow-ii-horizontal-frame-fields}
For every $(p,q)$-tensor $U$,
\[
 \nabla_c\!\left(U^{a_1\cdots a_q}_{b_1\cdots b_p}\right)
 =(\nabla_cU)^{a_1\cdots a_q}_{b_1\cdots b_p}.
\]
Thus the global fields $\nabla_a$ realize the covariant derivative of tensors
as ordinary directional derivatives of their frame components.
\end{lemma}

\section{Commutators and the orthonormal bundle}

\begin{lemma}[Frame-bundle commutators]
\label{lem:chow-ii-frame-bundle-commutators}
\dcref{appf:F.16,appf:F.17}
\group{Frame Bundle Calculus}
\source{chow-et-al-ricci-flow-part-ii:page-0449,chow-et-al-ricci-flow-part-ii:page-0450}
\uses{def:chow-ii-horizontal-frame-fields,def:chow-ii-vertical-frame-fields}
On $\mathcal F\mathcal M$,
\[
 [\nabla_a,\nabla_b]=-R^d{}_{abc}\Lambda^c_d,
\quad [\Lambda^a_b,\nabla_c]=\delta^a_c\nabla_b,
\quad [\Lambda^a_b,\Lambda^c_d]=\delta^a_d\Lambda^c_b-\delta^c_b\Lambda^a_d.
\]
Consequently, for a $(p,q)$-tensor $U$,
\[
 [\nabla_c,\nabla_d]U^{a_1\cdots a_q}_{b_1\cdots b_p}
 =-\sum_{\ell=1}^pR^f{}_{cdb_\ell e}U^{a_1\cdots a_q}_{b_1\cdots f\cdots b_p}
 +\sum_{k=1}^qR^{a_k}{}_{cde}U^{a_1\cdots e\cdots a_q}_{b_1\cdots b_p}.
\]
\end{lemma}

\begin{definition}[Rotation fields]
\label{def:chow-ii-rotation-fields}
\dcref{appf:F.19}
\group{Orthonormal Frame Bundle}
\source{chow-et-al-ricci-flow-part-ii:page-0451}
\uses{def:chow-ii-vertical-frame-fields}
Define the vertical rotation fields
\[
 \rho_{ab}=g_{ac}\Lambda^c_b-g_{bc}\Lambda^c_a.
\]
On $\mathcal O\mathcal M$, this is $\rho_{ab}=\Lambda^a_b-\Lambda^b_a$ and
$\rho_{ba}=-\rho_{ab}$.
\end{definition}

\begin{lemma}[The orthonormal frame-bundle frame]
\label{lem:chow-ii-orthonormal-frame-bundle}
\dcref{appf:F.23,appf:F.24,appf:F.25}
\group{Orthonormal Frame Bundle}
\source{chow-et-al-ricci-flow-part-ii:page-0452,chow-et-al-ricci-flow-part-ii:page-0453}
\uses{def:chow-ii-horizontal-frame-fields,def:chow-ii-rotation-fields}
The fields $\rho_{ab}$ are tangent to $\mathcal O\mathcal M$ and act on a
1-form by $\rho_{ab}(V_c)=g_{ac}V_b-g_{bc}V_a$.  The fields
$\{\nabla_a\}$ and $\{\rho_{bc}\}_{b<c}$ form an orthogonal basis on
$\mathcal O\mathcal M$, with $|\nabla_a|=1$ and $|\rho_{bc}|=\sqrt2$.  Moreover,
\[
 [\nabla_a,\nabla_b]=\tfrac12R_{abcd}g^{ce}\rho_{de},
 \qquad
 [\rho_{ab},\nabla_c]=g_{ac}\nabla_b-g_{bc}\nabla_a,
\]
and
\[
 [\rho_{ab},\rho_{cd}]=g_{ac}\rho_{bd}+g_{bd}\rho_{ac}
 -g_{ad}\rho_{bc}-g_{bc}\rho_{ad}.
\]
\end{lemma}

\section{Time-dependent frames and heat commutators}

\begin{definition}[The time vector field]
\label{def:chow-ii-frame-time-vector}
\dcref{appf:F.29}
\group{Ricci-Flow Frame Bundle}
\source{chow-et-al-ricci-flow-part-ii:page-0454}
\uses{def:chow-ii-vertical-frame-fields}
For a Ricci flow $g(t)$, define on $\mathcal F\mathcal M\times[0,T)$
\[
 \nabla_t=\frac{\partial}{\partial t}+R_{ab}g^{bc}\Lambda^a_c.
\]
Its restriction to the evolving orthonormal frame bundle is the unique vector
field tangent to that bundle whose spatial correction is vertical and normal
to each time slice $\mathcal O\mathcal M_{g(t)}$.
\end{definition}

\begin{lemma}[Time covariant derivative of tensor components]
\label{lem:chow-ii-time-covariant-derivative}
\dcref{appf:F.33}
\group{Ricci-Flow Frame Bundle}
\source{chow-et-al-ricci-flow-part-ii:page-0455}
\uses{def:chow-ii-frame-time-vector,lem:chow-ii-vertical-tensor-action}
For a time-dependent $(p,q)$-tensor,
\[
\begin{aligned}
 \nabla_t U^{a_1\cdots a_q}_{b_1\cdots b_p}
 ={}&\partial_tU^{a_1\cdots a_q}_{b_1\cdots b_p}
 +\sum_{j=1}^pR_{b_jc}g^{cd}U^{a_1\cdots a_q}_{b_1\cdots d\cdots b_p}\\
 &-\sum_{i=1}^qg^{a_ic}R_c{}^dU^{a_1\cdots d\cdots a_q}_{b_1\cdots b_p}.
\end{aligned}
\]
In particular, $\nabla_tg_{ab}=0$.
\end{lemma}

\begin{lemma}[Curvature evolution in the moving frame]
\label{lem:chow-ii-frame-curvature-evolution}
\dcref{appf:eq-F.24,appf:eq-F.25}
\group{Ricci-Flow Frame Bundle}
\source{chow-et-al-ricci-flow-part-ii:page-0456,chow-et-al-ricci-flow-part-ii:page-0457}
\uses{def:chow-ii-frame-time-vector}
On $\mathcal O\mathcal M$, with $B_{abcd}=-R_{aebf}R_{cedf}$,
\[
 (\nabla_t-\Delta)R_{abcd}
 =2(B_{abcd}-B_{abdc}+B_{acbd}-B_{adbc}),
 \qquad
 (\nabla_t-\Delta)R_{ab}=2R_{acdb}R_{cd},
\]
where $\Delta=\sum_e\nabla_e\nabla_e$.
\end{lemma}

\begin{theorem}[Heat and covariant-derivative commutators]
\label{thm:chow-ii-frame-heat-commutators}
\dcref{appf:F.35}
\group{Ricci-Flow Frame Bundle}
\source{chow-et-al-ricci-flow-part-ii:page-0457,chow-et-al-ricci-flow-part-ii:page-0458}
\uses{lem:chow-ii-orthonormal-frame-bundle,def:chow-ii-frame-time-vector}
On $\mathcal O\mathcal M$,
\[
 [\nabla_t,\nabla_a]=\nabla_bR_{ac}\rho_{bc}+R_{ac}\nabla_c,
\]
\[
 [\Delta,\nabla_a]=R_{eadc}\nabla_e\rho_{cd}+R_{ad}\nabla_d
 +\nabla_bR_{ac}\rho_{bc},
\]
and therefore
\[
 (\nabla_t-\Delta)\nabla_a-
 \nabla_a(\nabla_t-\Delta)=R_{abdc}\nabla_b\rho_{cd}.
\]
For a 1-form, 2-tensor, and 3-tensor respectively this gives
\[
\begin{aligned}
 [(\nabla_t-\Delta),\nabla_a]V_b&=2R_{acdb}\nabla_cV_d,\\
 [(\nabla_t-\Delta),\nabla_a]V_{bc}&=2R_{adeb}\nabla_dV_{ec}+2R_{adec}\nabla_dV_{be},\\
 [(\nabla_t-\Delta),\nabla_a]V_{bcd}&=R_{aefb}\nabla_eV_{fcd}+R_{aefc}\nabla_eV_{bfd}+R_{aefd}\nabla_eV_{bcf}.
\end{aligned}
\]
\end{theorem}
