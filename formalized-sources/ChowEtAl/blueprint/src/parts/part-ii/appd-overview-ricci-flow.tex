\section{Existence, uniqueness, convergence, and curvature evolution}

\begin{theorem}[Short time existence---closed manifolds]
\label{thm:chow-ii-appd-short-time-existence-closed}
\dcref{appD:D.1}
\group{Ricci Flow Existence}
\source{chow-et-al-ricci-flow-part-ii:page-0384}
If $(\mathcal M^n,g_0)$ is a smooth closed Riemannian manifold, there is a
unique smooth Ricci flow $g(t)$, $t\in[0,T)$, with $g(0)=g_0$.
\end{theorem}

\begin{theorem}[Short time existence---complete with bounded curvature]
\label{thm:chow-ii-appd-short-time-existence-complete}
\dcref{appD:D.2}
\group{Ricci Flow Existence}
\source{chow-et-al-ricci-flow-part-ii:page-0384}
If $(\mathcal M^n,g_0)$ is complete, noncompact, and has bounded sectional
curvature, then it admits a complete Ricci flow $g(t)$, $t\in[0,T)$, with
$g(0)=g_0$ and sectional curvature bounded on compact time intervals.
\end{theorem}

\begin{theorem}[Uniqueness---complete with bounded curvature]
\label{thm:chow-ii-appd-uniqueness-complete}
\dcref{appD:D.3}
\group{Ricci Flow Existence}
\source{chow-et-al-ricci-flow-part-ii:page-0384}
Let $g_i(t)$, $t\in[0,T_i)$, for $i=1,2$, be complete Ricci flows with
bounded sectional curvature on a noncompact manifold. If $g_1(0)=g_2(0)$,
then $g_1(t)=g_2(t)$ for $t<\min\{T_1,T_2\}$.
\end{theorem}

\begin{theorem}[Long time existence---closed manifolds]
\label{thm:chow-ii-appd-long-time-existence-closed}
\dcref{appD:D.4}
\group{Ricci Flow Existence}
\source{chow-et-al-ricci-flow-part-ii:page-0385}
\uses{thm:chow-ii-appd-short-time-existence-closed}
The Ricci flow from a smooth metric on a closed manifold has a unique maximal
solution on $[0,T)$, where $T\leq\infty$. If $T<\infty$, then
\[
 \lim_{t\nearrow T}\sup_{x\in\mathcal M}|\operatorname{Rm}(x,t)|=\infty.
\]
\end{theorem}

\begin{theorem}[Long time existence---complete with bounded curvature]
\label{thm:chow-ii-appd-long-time-existence-complete}
\dcref{appD:D.5}
\group{Ricci Flow Existence}
\source{chow-et-al-ricci-flow-part-ii:page-0385}

In the class of complete solutions with bounded sectional curvature, a
complete noncompact initial metric of bounded sectional curvature has a unique
maximal Ricci flow on $[0,T)$. If $T<\infty$, then
\[
 \lim_{t\nearrow T}\sup_{x\in\mathcal M}|\operatorname{Rm}(x,t)|=\infty.
\]
\end{theorem}

\begin{theorem}[Long time existence improved]

\begin{theorem}[Hamilton, 3-manifolds with positive Ricci curvature]
\label{thm:chow-ii-appd-hamilton-positive-ricci-three-manifolds}

\begin{theorem}[Closed manifolds with 2-positive curvature operator]

\begin{definition}[Curvature-operator evolution]
\label{def:chow-ii-appd-curvature-operator-evolution}
\dcref{appD:D.1-curvature-evolution}
\group{Curvature Evolution}
\source{chow-et-al-ricci-flow-part-ii:page-0386,chow-et-al-ricci-flow-part-ii:page-0387}
On $\Lambda^2T_x^*\mathcal M$, define
\[
 [U,V]_{ij}=g^{k\ell}(U_{ik}V_{\ell j}-V_{ik}U_{\ell j}).
\]
If $B_{abcd}=-R_{aebf}R_{cedf}$, then
\[
 (\operatorname{Rm}^{\#})_{ijkl}=2(B_{ikj\ell}-B_{iljk}),\qquad
 (\operatorname{Rm}^{2})_{ijkl}=2(B_{ijk\ell}-B_{ij\ell k}),
\]
and Ricci flow satisfies
\[
 \partial_t\operatorname{Rm}=\Delta\operatorname{Rm}
   +\operatorname{Rm}^{2}+\operatorname{Rm}^{\#},\qquad
 \partial_tR_{ab}=\Delta R_{ab}+2R_{acdb}R_{cd}.
\]
\end{definition}

\section{The rotationally symmetric neckpinch}

\begin{definition}[Rotationally symmetric Ricci flow]
\label{def:chow-ii-appd-rotationally-symmetric-flow}
\dcref{appD:D.6--D.8}
\group{Rotational Neckpinches}
\source{chow-et-al-ricci-flow-part-ii:page-0387,chow-et-al-ricci-flow-part-ii:page-0388}
On $S^n\setminus\{P_-,P_+\}\cong(-1,1)\times S^{n-1}$, write
\[
 g=h(r)^2dr^2+f(r)^2g_{S^{n-1}},\qquad ds=h(r)\,dr.
\]
Then
\[
 K_{\rm rad}=-\frac{f''}{f},\qquad
 K_{\rm sph}=\frac{1-(f')^2}{f^2},
\]
and Ricci flow is equivalent to
\[
 \partial_th=(n-1)\frac{f''}{f}h,\qquad
 \partial_tf=f''-(n-2)\frac{1-(f')^2}{f}.
\]
Moreover
$[\partial_t,\partial_s]=(n-1)K_{\rm rad}\partial_s$.
\end{definition}

\begin{theorem}[Existence of rotationally symmetric neckpinch]
\label{thm:chow-ii-appd-neckpinch-existence}
\dcref{appD:D.11}
\group{Rotational Neckpinches}
\source{chow-et-al-ricci-flow-part-ii:page-0388,chow-et-al-ricci-flow-part-ii:page-0389}
\uses{def:chow-ii-appd-rotationally-symmetric-flow}
There is an open set of rotationally symmetric metrics on $S^n$ having a
sufficiently pinched neck, positive scalar and spherical sectional curvature,
and positive Ricci curvature on both polar caps, whose Ricci flows develop a
finite-time Type I neckpinch. If $\bar s(t)$ is the smallest neck, then in
\[
 |s-\bar s|\leq2\sqrt{(T-t)\log(1/(T-t))}
\]
one has, uniformly as $t\nearrow T$,
\[
 1+o(1)\leq\frac{f}{\sqrt{2(n-2)(T-t)}}
 \leq1+C\frac{(s-\bar s)^2}{(T-t)\log(1/(T-t))},
\]
and the corresponding intermediate-region estimate is
\[
 \frac f{\sqrt{T-t}}\leq
 C\frac{s-\bar s}{\sqrt{(T-t)\log(1/(T-t))}}
 \sqrt{\log\left(\frac{|s-\bar s|}
 {\sqrt{(T-t)\log(1/(T-t))}}\right)}.
\]
\end{theorem}

\begin{theorem}[Asymptotics of rotationally symmetric neckpinch]
\label{thm:chow-ii-appd-neckpinch-asymptotics}
\dcref{appD:D.12}
\group{Rotational Neckpinches}
\source{chow-et-al-ricci-flow-part-ii:page-0390}
\uses{thm:chow-ii-appd-neckpinch-existence}
Under the preceding hypotheses and reflection symmetry about the smallest
neck, the singular time depends continuously on the initial metric, the
diameter stays bounded, and the singularity occurs only on
$\{0\}\times S^{n-1}$. With
\[
 u=\frac f{\sqrt{2(n-2)(T-t)}},\quad
 \sigma=\frac s{\sqrt{T-t}},\quad \tau=\log\frac1{T-t},
\]
the inner and intermediate profiles satisfy
\[
 u=1+\frac{\sigma^2-2}{8\tau}+o(\tau^{-1}),\qquad
 u=\sqrt{1+(1+o(1))\frac{\sigma^2}{4\tau}},
\]
uniformly on the respective regions stated in the source. In the outer
region, for every $\varepsilon>0$, eventually
\[
 \left(\frac{\sqrt{n-2}}2-\varepsilon\right)
 \frac{s}{\sqrt{\log(1/s)}}\leq f\leq
 \left(\frac{\sqrt{n-2}}2+\varepsilon\right)
 \frac{s}{\sqrt{\log(1/s)}}.
\]
\end{theorem}

\begin{remark}[Projective neckpinch]
\label{rem:chow-ii-appd-projective-neckpinch}
\dcref{appD:D.13}

\begin{remark}[Growing normalized neck length]

\begin{proposition}[Estimates for neckpinches]
\label{prop:chow-ii-appd-neckpinch-estimates}
\dcref{appD:D.16}
\group{Rotational Neckpinches}
\source{chow-et-al-ricci-flow-part-ii:page-0390,chow-et-al-ricci-flow-part-ii:page-0391}
\uses{def:chow-ii-appd-rotationally-symmetric-flow,thm:chow-ii-appd-neckpinch-existence}
For a flow in Theorem~\ref{thm:chow-ii-appd-neckpinch-existence}, the gradient
bound $|f'|\leq1$ is preserved; $a=f^2(K_{\rm sph}-K_{\rm rad})$ is uniformly
bounded; $|\operatorname{Rm}|\leq C/f^2$; $f$ decreases pointwise and $f^2$
is uniformly Lipschitz in time; and
\[
 (n-2)(T-t)\leq f_{\min}(t)^2\leq2(n-2)(T-t),\qquad
 |\operatorname{Rm}|\leq\frac C{T-t}.
\]
The number of zeros of $f'$ is finite and nonincreasing, the polar caps remain
concave, and
\[
 -\frac{K_{\rm rad}}{K_{\rm sph}}\log(K_{\rm sph}+c)
 \leq\max\left\{n-2,
 \sup_{t=0}-\frac{K_{\rm rad}}{K_{\rm sph}}
 \log(K_{\rm sph}+c)\right\},
\]
where $c=2-\log(\inf K_{\rm sph}(\cdot,0))$.
\end{proposition}

\begin{theorem}[Hamilton--Ivey estimate]
\label{thm:chow-ii-appd-hamilton-ivey}
\dcref{appD:D.17}
\group{Curvature Pinching}
\source{chow-et-al-ricci-flow-part-ii:page-0392}

$\lambda_{\min}(\operatorname{Rm})<0$,
\[
 R\geq|\lambda_{\min}|\bigl(\log|\lambda_{\min}|+log(1+t)-3\bigr),
\]
and hence $R\geq|\lambda_{\min}|(\log|\lambda_{\min}|-3)$.
\end{theorem}

\begin{corollary}[A 3-dimensional solution has $\phi$-almost nonnegative curvature]
\label{cor:chow-ii-appd-almost-nonnegative-curvature}

\begin{theorem}[Strong maximum principle for $\operatorname{Rm}$]
\label{thm:chow-ii-appd-curvature-operator-strong-maximum-principle}
\dcref{appD:D.21}
\group{Curvature Rigidity}
\source{chow-et-al-ricci-flow-part-ii:page-0393}
If a Ricci flow has nonnegative curvature operator, then for some $\delta>0$
and each $t\in(0,\delta)$, the image of $\operatorname{Rm}[g(t)]$ is a smooth
subbundle of $\Lambda^2T^*\mathcal M$ which is invariant under parallel

\end{theorem}

\begin{theorem}[3-manifolds with $\operatorname{Rm}\geq0$ or $\operatorname{Ric}\geq0$]
\label{thm:chow-ii-appd-three-manifold-curvature-rigidity}
\dcref{appD:D.22}
\group{Curvature Rigidity}
\source{chow-et-al-ricci-flow-part-ii:page-0393}
\uses{thm:chow-ii-appd-curvature-operator-strong-maximum-principle}
For a Ricci flow on a closed three-manifold with nonnegative sectional,
respectively Ricci, curvature, its universal covering flow at each positive
time is either flat $\mathbb R^3$, a product $(S^2,h(t))\times\mathbb R$ with
$h(t)$ positively curved, or has positive sectional, respectively Ricci,
curvature. In the last case the universal cover is diffeomorphic to $S^3$ and
the original manifold is a spherical space form.
\end{theorem}

\begin{theorem}[Trace Harnack estimate for Ricci flow]
\label{thm:chow-ii-appd-trace-harnack}
\dcref{appD:D.26}
\group{Harnack Estimates}
\source{chow-et-al-ricci-flow-part-ii:page-0394,chow-et-al-ricci-flow-part-ii:page-0395}
\uses{cor:chow-ii-trace-harnack-and-tr-monotonicity}
For a compact, or complete bounded-curvature, Ricci flow with nonnegative
curvature operator, every vector field $V$ satisfies
\[
 \partial_tR+\frac Rt+2\langle\nabla R,V\rangle
 +2\operatorname{Ric}(V,V)\geq0.
\]
In particular $\partial_t(tR)\geq0$ and
$R(x,t_2)\geq(t_1/t_2)R(x,t_1)$ when $0<t_1\leq t_2$.
\end{theorem}

\begin{theorem}[Matrix Harnack estimate for Ricci flow]
\label{thm:chow-ii-appd-matrix-harnack}
\dcref{appD:D.27}
\group{Harnack Estimates}
\source{chow-et-al-ricci-flow-part-ii:page-0395}
\uses{thm:chow-ii-hamilton-matrix-harnack}
For a complete bounded-curvature Ricci flow with nonnegative curvature
operator, Hamilton's Harnack quadratic satisfies
\[
 M_{ij}W^iW^j+2P_{pij}U^{pi}W^j+R_{pijq}U^{pi}U^{qj}\geq0
\]
for every one-form $W$ and two-form $U$.
\end{theorem}

\begin{theorem}[Linear trace Harnack estimate]
\label{thm:chow-ii-appd-linear-trace-harnack}
\dcref{appD:D.28}
\group{Harnack Estimates}
\source{chow-et-al-ricci-flow-part-ii:page-0395,chow-et-al-ricci-flow-part-ii:page-0396}

$g(t)$ is complete with bounded nonnegative curvature operator, $h(0)\geq0$,
and $h$ obeys the exponential initial growth and Gaussian space-time

\[
 \operatorname{div}(\operatorname{div}h)+\langle\operatorname{Ric},h\rangle
 +2\langle\operatorname{div}h,V\rangle+h(V,V)+\frac{\operatorname{tr}h}{2t}

\]
\end{theorem}

\section{Perelman's energy, entropy, and associated invariants}

\begin{definition}[Perelman's energy and entropy]
\label{def:chow-ii-appd-perelman-energy-entropy}
\dcref{appD:D.34--D.43}
\group{Perelman Functionals}
\source{chow-et-al-ricci-flow-part-ii:page-0396,chow-et-al-ricci-flow-part-ii:page-0397}
For a closed Riemannian manifold, define
\[
 \mathcal F(g,f)=\int_{\mathcal M}(R+|\nabla f|^2)e^{-f}\,d\mu
\]
and, for $\tau>0$,
\[
 \mathcal W(g,f,\tau)=\int_{\mathcal M}
 [\tau(R+|\nabla f|^2)+f-n](4\pi\tau)^{-n/2}e^{-f}\,d\mu.
\]
If $\partial_tg=-2\operatorname{Ric}$ and
$\partial_tf=-\Delta f+|\nabla f|^2-R$, then
\[
 \frac d{dt}\mathcal F=2\int_{\mathcal M}
 |\operatorname{Ric}+\nabla^2f|^2e^{-f}\,d\mu.
\]
If instead $\partial_tf=-\Delta f+|\nabla f|^2-R+n/(2\tau)$ and
$\tau'=-1$, then
\[
 \frac d{dt}\mathcal W=2\tau\int_{\mathcal M}
 \left|\operatorname{Ric}+\nabla^2f-\frac g{2\tau}\right|^2
 (4\pi\tau)^{-n/2}e^{-f}\,d\mu.
\]
\end{definition}

\begin{definition}[$\lambda$-, $\bar\lambda$-, $\mu$-, and $\nu$-invariants]
\label{def:chow-ii-appd-perelman-invariants}
\dcref{appD:D.43--D.50,D.53}
\group{Perelman Functionals}
\source{chow-et-al-ricci-flow-part-ii:page-0397,chow-et-al-ricci-flow-part-ii:page-0398,chow-et-al-ricci-flow-part-ii:page-0399,chow-et-al-ricci-flow-part-ii:page-0401}
Define
\[
 \lambda(g)=\inf_{\int e^{-f}d\mu=1}\mathcal F(g,f),\qquad
 \bar\lambda(g)=\lambda(g)\operatorname{Vol}(\mathcal M,g)^{2/n},
\]
\[
 \mu(g,\tau)=\inf_{\int(4\pi\tau)^{-n/2}e^{-f}d\mu=1}
 \mathcal W(g,f,\tau),\qquad
 \nu(g)=\inf_{\tau>0}\mu(g,\tau).
\]
Equivalently, $\lambda(g)$ is the lowest eigenvalue of $-4\Delta+R$ and
has a unique positive normalized eigenfunction. The infimum defining
$\mu(g,\tau)$ is finite and is attained by a smooth minimizer.
\end{definition}

\begin{lemma}[$\lambda$-invariant monotonicity]
\label{lem:chow-ii-appd-lambda-monotonicity}
\dcref{appD:D.29}
\group{Perelman Invariants}
\source{chow-et-al-ricci-flow-part-ii:page-0398}
\uses{def:chow-ii-appd-perelman-energy-entropy,def:chow-ii-appd-perelman-invariants}
Along a Ricci flow on a closed manifold,
\[
 \frac d{dt}\lambda(g(t))\geq\frac2n\lambda(g(t))^2
\]
in the sense of the lower limit of backward difference quotients. In
particular, $\lambda(g(t))$ is nondecreasing.
\end{lemma}

\begin{lemma}[No nontrivial steady breathers on closed manifolds]
\label{lem:chow-ii-appd-no-steady-breathers}
\dcref{appD:D.30}
\group{Perelman Invariants}
\source{chow-et-al-ricci-flow-part-ii:page-0399}
\uses{lem:chow-ii-appd-lambda-monotonicity}
If a closed Ricci flow satisfies $\lambda(g(t_1))=\lambda(g(t_2))$ for
$t_1<t_2$, then it is Ricci-flat. Thus every steady Ricci breather on a
closed manifold is Ricci-flat.
\end{lemma}

\begin{lemma}[$\bar\lambda$-invariant monotonicity when $\bar\lambda\leq0$]
\label{lem:chow-ii-appd-normalized-lambda-monotonicity}
\dcref{appD:D.31}
\group{Perelman Invariants}
\source{chow-et-al-ricci-flow-part-ii:page-0399}
\uses{def:chow-ii-appd-perelman-invariants,lem:chow-ii-appd-lambda-monotonicity}
At any time at which $\bar\lambda(g(t))\leq0$,
\[
 \frac d{dt}\bar\lambda(g(t))\geq
 2\operatorname{Vol}(g(t))^{2/n}\int_{\mathcal M}
 \left|\operatorname{Ric}+\nabla^2f-
 \frac{R+\Delta f}{n}g\right|^2e^{-f}\,d\mu\geq0,
\]
where $f$ minimizes $\mathcal F$. Equality implies that $g(t)$ is a gradient
Ricci soliton.
\end{lemma}

\begin{lemma}[No nontrivial expanding breathers]
\label{lem:chow-ii-appd-no-expanding-breathers}
\dcref{appD:D.32}
\group{Perelman Invariants}
\source{chow-et-al-ricci-flow-part-ii:page-0399}
\uses{lem:chow-ii-appd-normalized-lambda-monotonicity}
Every expanding or steady Ricci breather on a closed manifold is Einstein.
\end{lemma}

\begin{lemma}[Monotonicity formula for the classical entropy]
\label{lem:chow-ii-appd-classical-entropy-monotonicity}
\dcref{appD:D.33}
\group{Perelman Invariants}
\source{chow-et-al-ricci-flow-part-ii:page-0399}
\uses{def:chow-ii-appd-perelman-energy-entropy}
Suppose $(g(t),f(t))$ obeys the coupled energy evolution on a closed manifold,
and set $\mathcal N=\int f e^{-f}\,d\mu$. Then
\[
 \mathcal F(g(t),f(t))\leq\frac n{2(T-t)}\int e^{-f}\,d\mu
\]
and
\[
 \frac d{dt}\left(\mathcal N-
 \frac n2\left(\int e^{-f}\,d\mu\right)\log(T-t)\right)\geq0.
\]
\end{lemma}

\section{Compactness, no local collapsing, and singularity models}

\begin{theorem}[Compactness for solutions]
\label{thm:chow-ii-appd-ricci-flow-compactness}
\dcref{appD:D.34}
\group{Ricci Flow Compactness}
\source{chow-et-al-ricci-flow-part-ii:page-0400}
Let $(\mathcal M_k,g_k(t),O_k)$ be complete pointed Ricci flows on a common
interval $(\alpha,\omega)$ containing zero. If their curvatures are uniformly
bounded and $\operatorname{inj}_{g_k(0)}(O_k)\geq\iota_0>0$, then a subsequence
converges in the smooth pointed Cheeger--Gromov sense to a complete pointed
Ricci flow on $(\alpha,\omega)$.
\end{theorem}

\begin{remark}[Compactness without injectivity-radius bound]
\label{rem:chow-ii-appd-compactness-without-injectivity}
\dcref{appD:D.35}
\group{Ricci Flow Compactness}
\source{chow-et-al-ricci-flow-part-ii:page-0400}
There is an analogue of Theorem~\ref{thm:chow-ii-appd-ricci-flow-compactness}
in a broader category which does not assume a lower injectivity-radius bound.
\end{remark}

\begin{theorem}[Perelman---no local collapsing]
\label{thm:chow-ii-appd-no-local-collapsing}
\dcref{appD:D.36}
\group{No Local Collapsing}
\source{chow-et-al-ricci-flow-part-ii:page-0400}
For a Ricci flow $g(t)$ on a closed manifold with finite maximal time $T$ and
every $\rho>0$, there is $\kappa=\kappa(g(0),T,\rho)>0$ such that $g(t)$ is
$\kappa$-noncollapsed below scale $\rho$ for every $t<T$.
\end{theorem}

\begin{theorem}[No local collapsing theorem improved]
\label{thm:chow-ii-appd-improved-no-local-collapsing}
\dcref{appD:D.37}
\group{No Local Collapsing}
\source{chow-et-al-ricci-flow-part-ii:page-0401}
\uses{thm:chow-ii-appd-no-local-collapsing}
In the preceding finite-time closed setting, for every $\rho>0$ there is
$\kappa>0$ such that if $r\leq\rho$ and $R\leq r^{-2}$ on
$B_{g(t)}(p,r)$, then
\[
 \frac{\operatorname{Vol}_{g(t)}B_{g(t)}(p,s)}{s^n}\geq\kappa
 \qquad(0<s\leq r).
\]
\end{theorem}

\begin{lemma}[$\mu$-invariant monotonicity]
\label{lem:chow-ii-appd-mu-monotonicity}
\dcref{appD:D.38}
\group{Perelman Invariants}
\source{chow-et-al-ricci-flow-part-ii:page-0401}
\uses{def:chow-ii-appd-perelman-energy-entropy,def:chow-ii-appd-perelman-invariants}
If $g(t)$ is a closed Ricci flow and $\tau'(t)=-1$ with $\tau>0$, then
\[
 \mu(g(t_2),\tau(t_2))\geq\mu(g(t_1),\tau(t_1))
 \qquad(0\leq t_1\leq t_2).
\]
\end{lemma}

\begin{lemma}[$\mu$ under Cheeger--Gromov convergence]
\label{lem:chow-ii-appd-mu-cheeger-gromov}
\dcref{appD:D.39}
\group{Perelman Invariants}
\source{chow-et-al-ricci-flow-part-ii:page-0402}
\uses{def:chow-ii-appd-perelman-invariants}
If $(\mathcal N_k,g_k,x_k)$ converges smoothly in the pointed
Cheeger--Gromov sense to $(\mathcal N_\infty,g_\infty,x_\infty)$, then for
every $\tau>0$,
\[
 \mu(g_\infty,\tau)\geq\limsup_{k\to\infty}\mu(g_k,\tau).
\]
\end{lemma}

\begin{lemma}[$\mu\to\infty$ as $\tau\to\infty$ when $\lambda>0$]
\label{lem:chow-ii-appd-mu-large-tau}

\group{Perelman Invariants}

If $\lambda(g)>0$, then $\lim_{\tau\to\infty}\mu(g,\tau)=\infty$.
\end{lemma}

\begin{lemma}[Behavior of $\mu(g,\tau)$ for $\tau$ small]
\label{lem:chow-ii-appd-mu-small-tau}
\dcref{appD:D.42}
\group{Perelman Invariants}
\source{chow-et-al-ricci-flow-part-ii:page-0402}
\uses{def:chow-ii-appd-perelman-invariants}
For every closed Riemannian manifold there is $\bar\tau>0$ such that
$\mu(g,\tau)<0$ for $0<\tau<\bar\tau$, and
$\lim_{\tau\downarrow0}\mu(g,\tau)=0$.
\end{lemma}

\begin{lemma}[$\nu$-invariant monotonicity]
\label{lem:chow-ii-appd-nu-monotonicity}
\dcref{appD:D.43}
\group{Perelman Invariants}
\source{chow-et-al-ricci-flow-part-ii:page-0402}
\uses{def:chow-ii-appd-perelman-invariants,lem:chow-ii-appd-mu-monotonicity}
Along a closed Ricci flow, $\nu(g(t))$ is nondecreasing whenever it is finite
and defined. If $\lambda>0$ and $\nu$ is constant on an interval, the flow is
a gradient shrinking soliton. If $\nu(g(t_0))=-\infty$, then
$\nu(g(t))=-\infty$ for every $t\leq t_0$.
\end{lemma}

\begin{proposition}[$\mu$ controls volume ratios]
\label{prop:chow-ii-appd-mu-controls-volume-ratios}
\dcref{appD:D.44}
\group{Volume Ratios}
\source{chow-et-al-ricci-flow-part-ii:page-0402,chow-et-al-ricci-flow-part-ii:page-0403}
\uses{def:chow-ii-appd-perelman-invariants}
For every $\rho>0$ there is $C_2(n,\rho)$ such that, if $r\leq\rho$ and
$\operatorname{Ric}\geq-c_1(n)r^{-2}$ and $R\leq c_1(n)r^{-2}$ on
$B(p,r)$ in a closed Riemannian manifold $(\widehat{\mathcal M},\hat g)$,
then
\[
 \mu(\hat g,r^2)\leq
 \log\frac{\operatorname{Vol}B(p,r)}{r^n}+C_2(n,\rho).
\]
\end{proposition}

\begin{proposition}[Bounding $\mu$ by the scalar curvature and volume ratio]
\label{prop:chow-ii-appd-bound-mu-local-geometry}
\dcref{appD:D.45}
\group{Volume Ratios}
\source{chow-et-al-ricci-flow-part-ii:page-0403}
\uses{def:chow-ii-appd-perelman-invariants}
For every ball $B(p,r)$ in a closed Riemannian manifold,
\[
 \mu(\hat g,r^2)\leq\log\frac{\operatorname{Vol}B(p,r)}{r^n}
 +\left(36+\frac{r^2\int_{B(p,r)}R_+\,d\mu}
 {\operatorname{Vol}B(p,r)}\right)
 \frac{\operatorname{Vol}B(p,r)}{\operatorname{Vol}B(p,r/2)},
\]
where $R_+=\max\{R,0\}$.
\end{proposition}

\begin{proposition}[Bounding volume ratios by $\nu_r$]
\label{prop:chow-ii-appd-bound-volume-ratios-nu}
\dcref{appD:D.46}
\group{Volume Ratios}
\source{chow-et-al-ricci-flow-part-ii:page-0403}
\uses{def:chow-ii-appd-perelman-invariants}
If $R\leq c_1(n)r^{-2}$ on $B(p,r)$ in a closed Riemannian manifold and
$0<s\leq r$, then
\[
 \frac{\operatorname{Vol}B(p,s)}{s^n}\geq
 e^{-3^n(36+c_1(n))}e^{\nu_r(\hat g)},
 \qquad
 \nu_r(\hat g):=\inf_{0<\tau\leq r^2}\mu(\hat g,\tau)\geq\nu(\hat g).
\]
\end{proposition}

\begin{corollary}[Obtaining singularity models]
\label{cor:chow-ii-appd-obtaining-singularity-models}
\dcref{appD:D.47}
\group{Singularity Models}
\source{chow-et-al-ricci-flow-part-ii:page-0403,chow-et-al-ricci-flow-part-ii:page-0404}
\uses{thm:chow-ii-appd-ricci-flow-compactness,thm:chow-ii-appd-no-local-collapsing}
Let a closed Ricci flow become singular at $T<\infty$, and let
$(x_i,t_i)\to T$ satisfy $K_i=|\operatorname{Rm}(x_i,t_i)|\to\infty$ and
\[
 \sup_{\mathcal M\times[t_i-\beta K_i^{-1},t_i]}
 |\operatorname{Rm}|\leq CK_i.
\]
After passing to a subsequence, the pointed dilations
\[
 g_i(t)=K_i g(t_i+K_i^{-1}t)
\]
converge on $[-\beta,0]$ to a complete bounded-curvature Ricci flow, possibly
on a different noncompact manifold. The limit is $\kappa$-noncollapsed on all
scales for a uniform $\kappa>0$.
\end{corollary}

\begin{corollary}[The cigar is not a finite time singularity model]
\label{cor:chow-ii-appd-cigar-not-singularity-model}
\dcref{appD:D.48}
\group{Singularity Models}
\source{chow-et-al-ricci-flow-part-ii:page-0404}
\uses{cor:chow-ii-appd-obtaining-singularity-models}
The cigar soliton, and its product with any flat Ricci flow such as the line,
cannot arise as a finite-time singularity model of a Ricci flow on a closed
manifold.
\end{corollary}

\begin{theorem}[Immortal solutions are expanding gradient solitons]
\label{thm:chow-ii-appd-immortal-expanding-soliton}
\dcref{appD:D.49}
\group{Singularity Models}
\source{chow-et-al-ricci-flow-part-ii:page-0404}
\uses{thm:chow-ii-appd-trace-harnack}
Let $(\mathcal M,g(t))$, $t\in(\alpha,\infty)$, be a complete immortal Ricci
flow with nonnegative curvature operator. If $tR(x,t)$ attains its space-time
maximum and $\mathcal M$ is simply connected with positive Ricci curvature,
then the flow is an expanding gradient Ricci soliton.
\end{theorem}
